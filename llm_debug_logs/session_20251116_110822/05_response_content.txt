    rsi_value = close.rsi(rsi_period)
    sma_value = close.sma(sma_period)

    # 3. Generate signals
    buy_signal = rsi_value < 30 and close > sma_value
    sell_signal = rsi_value > 70 and close < sma_value

    # 4. Plot results
    plot(buy_signal, "Buy", color=color.green)
